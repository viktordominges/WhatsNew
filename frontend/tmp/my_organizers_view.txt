// –ü–æ–¥—Å—á–µ—Ç –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–æ–≤ –ø–æ —Å—Ç–∞—Ç—É—Å–∞–º

    return {

        active: organizers.value.filter(o => o.status === 'published').length,
        pending: organizers.value.filter(o => o.status === 'pending').length

    error.value = null;

        // –ü–æ–ª—É—á–∞–µ–º –≤—Å–µ—Ö –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–æ–≤ –∏ —Ñ–∏–ª—å—Ç—Ä—É–µ–º –ø–æ —Ç–µ–∫—É—â–µ–º—É –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é

            o => o.author?.id === authStore.user?.id

        error.value = '–ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–≥—Ä—É–∑–∏—Ç—å —Å–ø–∏—Å–æ–∫ –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–æ–≤';
    } finally {
        loading.value = false;
    }
};

// –ü–µ—Ä–µ—Ö–æ–¥ –∫ —Å–æ–∑–¥–∞–Ω–∏—é –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞
const createOrganizer = () => {
    router.push('/dashboard/organizers/create');
};

// –ü–µ—Ä–µ—Ö–æ–¥ –∫ —Ä–µ–¥–∞–∫—Ç–∏—Ä–æ–≤–∞–Ω–∏—é

    router.push(`/dashboard/organizers/${slug}/edit`);
};

// –ü–µ—Ä–µ—Ö–æ–¥ –∫ —Å–æ–±—ã—Ç–∏—è–º –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞
const viewOrganizerEvents = (slug) => {
    router.push(`/dashboard/organizers/${slug}/events`);
};

// –£–¥–∞–ª–µ–Ω–∏–µ –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞
const deleteOrganizer = async (organizer) => {
    if (!confirm(`–í—ã —É–≤–µ—Ä–µ–Ω—ã —á—Ç–æ —Ö–æ—Ç–∏—Ç–µ —É–¥–∞–ª–∏—Ç—å –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞ "${organizer.name}"?`)) {
        return;
    }
    
    try {
        await organizersAPI.delete(organizer.slug);
        
        // –£–¥–∞–ª—è–µ–º –∏–∑ —Å–ø–∏—Å–∫–∞
        organizers.value = organizers.value.filter(o => o.slug !== organizer.slug);
        
        alert('–û—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä —É—Å–ø–µ—à–Ω–æ —É–¥–∞–ª–µ–Ω');
    } catch (err) {
        console.error('Error deleting organizer:', err);
        alert('–ù–µ —É–¥–∞–ª–æ—Å—å —É–¥–∞–ª–∏—Ç—å –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞');
    }
};

// –ü–æ–ª—É—á–∏—Ç—å —Ç–µ–∫—Å—Ç —Å—Ç–∞—Ç—É—Å–∞

    const statusMap = {
        'draft': '–ß–µ—Ä–Ω–æ–≤–∏–∫',
        'pending': '–ù–∞ –º–æ–¥–µ—Ä–∞—Ü–∏–∏',
        'published': '–ê–∫—Ç–∏–≤–µ–Ω',
        'rejected': '–û—Ç–∫–ª–æ–Ω–µ–Ω'
    };
    return statusMap[status] || status;
};

// –ü–æ–ª—É—á–∏—Ç—å –∫–ª–∞—Å—Å —Å—Ç–∞—Ç—É—Å–∞
const getStatusClass = (status) => {
    return `status-${status}`;
};

onMounted(() => {
    fetchOrganizers();
});
