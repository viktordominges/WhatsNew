import { ref, onMounted, computed } from 'vue';
import { useRoute, useRouter } from 'vue-router';
import { organizersAPI, activitiesAPI } from '@/services/api';

const route = useRoute();
const router = useRouter();

const organizer = ref(null);
const events = ref([]);
const loading = ref(true);
const error = ref(null);
const activeFilter = ref('all'); // all, published, pending, draft

// –§–∏–ª—å—Ç—Ä—ã
const filters = [
    { value: 'all', label: '–í—Å–µ', icon: 'üìã' },
    { value: 'published', label: '–ê–∫—Ç–∏–≤–Ω—ã–µ', icon: '‚úÖ' },
    { value: 'pending', label: '–ù–∞ –º–æ–¥–µ—Ä–∞—Ü–∏–∏', icon: '‚è≥' },
    { value: 'draft', label: '–ß–µ—Ä–Ω–æ–≤–∏–∫–∏', icon: 'üìù' }
];

// –û—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω–Ω—ã–µ —Å–æ–±—ã—Ç–∏—è
const filteredEvents = computed(() => {
    if (activeFilter.value === 'all') {
        return events.value;
    }
    return events.value.filter(e => e.status === activeFilter.value);
});

// –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
const stats = computed(() => {
    return {
        total: events.value.length,
        published: events.value.filter(e => e.status === 'published').length,
        pending: events.value.filter(e => e.status === 'pending').length,
        draft: events.value.filter(e => e.status === 'draft').length
    };
});

// –ó–∞–≥—Ä—É–∑–∫–∞ –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞
const fetchOrganizer = async () => {
    try {
        const response = await organizersAPI.getBySlug(route.params.slug);
        organizer.value = response.data;
    } catch (err) {
        console.error('Error fetching organizer:', err);
        error.value = '–ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–≥—Ä—É–∑–∏—Ç—å –¥–∞–Ω–Ω—ã–µ –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–∞';
    }
};

// –ó–∞–≥—Ä—É–∑–∫–∞ —Å–æ–±—ã—Ç–∏–π
const fetchEvents = async () => {
    try {
        const response = await organizersAPI.getActivities(route.params.slug);
        events.value = response.data;
    } catch (err) {
        console.error('Error fetching events:', err);
        error.value = '–ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–≥—Ä—É–∑–∏—Ç—å —Å–ø–∏—Å–æ–∫ —Å–æ–±—ã—Ç–∏–π';
    }
};

// –ó–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö
const loadData = async () => {
    loading.value = true;
    error.value = null;
    
    await Promise.all([
        fetchOrganizer(),
        fetchEvents()
    ]);
    
    loading.value = false;
};

// –°–æ–∑–¥–∞—Ç—å —Å–æ–±—ã—Ç–∏–µ
const createEvent = () => {
    router.push(`/dashboard/organizers/${route.params.slug}/events/create`);
};

// –†–µ–¥–∞–∫—Ç–∏—Ä–æ–≤–∞—Ç—å —Å–æ–±—ã—Ç–∏–µ
const editEvent = (eventSlug) => {
    router.push(`/dashboard/organizers/${route.params.slug}/events/${eventSlug}/edit`);
};

// –ü–æ—Å–º–æ—Ç—Ä–µ—Ç—å –Ω–∞ —Å–∞–π—Ç–µ
const viewEvent = (eventSlug) => {
    window.open(`/activities/${eventSlug}`, '_blank');
};

// –£–¥–∞–ª–∏—Ç—å —Å–æ–±—ã—Ç–∏–µ
const deleteEvent = async (event) => {
    if (!confirm(`–í—ã —É–≤–µ—Ä–µ–Ω—ã —á—Ç–æ —Ö–æ—Ç–∏—Ç–µ —É–¥–∞–ª–∏—Ç—å —Å–æ–±—ã—Ç–∏–µ "${event.name}"?`)) {
        return;
    }
    
    try {
        await activitiesAPI.delete(event.slug);
        events.value = events.value.filter(e => e.slug !== event.slug);
        alert('–°–æ–±—ã—Ç–∏–µ —É—Å–ø–µ—à–Ω–æ —É–¥–∞–ª–µ–Ω–æ');
    } catch (err) {
        console.error('Error deleting event:', err);
        alert('–ù–µ —É–¥–∞–ª–æ—Å—å —É–¥–∞–ª–∏—Ç—å —Å–æ–±—ã—Ç–∏–µ');
    }
};

// –í–µ—Ä–Ω—É—Ç—å—Å—è –∫ —Å–ø–∏—Å–∫—É –æ—Ä–≥–∞–Ω–∏–∑–∞—Ç–æ—Ä–æ–≤
const goBack = () => {
    router.push('/dashboard/organizers');
};

// –ü–æ–ª—É—á–∏—Ç—å —Ç–µ–∫—Å—Ç —Å—Ç–∞—Ç—É—Å–∞
const getStatusText = (status) => {
    const statusMap = {
        'draft': '–ß–µ—Ä–Ω–æ–≤–∏–∫',
        'pending': '–ù–∞ –º–æ–¥–µ—Ä–∞—Ü–∏–∏',
        'published': '–û–ø—É–±–ª–∏–∫–æ–≤–∞–Ω–æ',
        'rejected': '–û—Ç–∫–ª–æ–Ω–µ–Ω–æ',
        'cancelled': '–û—Ç–º–µ–Ω–µ–Ω–æ'
    };
    return statusMap[status] || status;
};

// –ü–æ–ª—É—á–∏—Ç—å –∫–ª–∞—Å—Å —Å—Ç–∞—Ç—É—Å–∞
const getStatusClass = (status) => {
    return `status-${status}`;
};

// –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –¥–∞—Ç—ã
const formatDate = (dateString) => {
    const date = new Date(dateString);
    return date.toLocaleDateString('ru-RU', { 
        day: 'numeric', 
        month: 'long', 
        year: 'numeric' 
    });
};

onMounted(() => {
    loadData();
});
</script>

                        —Ä–µ–¥–∞–∫—Ç–∏—Ä—É–π—Ç–µ —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–µ –∏ –æ—Ç—Å–ª–µ–∂–∏–≤–∞–π—Ç–µ –∏—Ö —Å—Ç–∞—Ç—É—Å.
